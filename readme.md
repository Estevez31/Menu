def draw_menu():
    oled.fill(0)
    for i, item in enumerate(menu_items):
        if i == selected_item:
            oled.text(">", 0, i * 10)
        oled.text(item, 10, i * 10)

# Función para navegar por el menú
def navigate_menu(direction):
    global selected_item
    if direction == "up":
        selected_item -= 1
        if selected_item < 0:
            selected_item = len(menu_items) - 1
    elif direction == "down":
        selected_item += 1
        if selected_item >= len(menu_items):
            selected_item = 0

# Función principal
def main():
    draw_menu()
